cmake_minimum_required(VERSION 3.14)
project(cpp_launch_test VERSION 0.1.0 LANGUAGES CXX)

# C++14 Standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Compiler warnings
if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic -Wno-unused-parameter)
endif()

# Find dependencies
find_package(Threads REQUIRED)

# Include directories
include_directories(
  ${CMAKE_CURRENT_SOURCE_DIR}/include
  ${CMAKE_CURRENT_SOURCE_DIR}/src
)

# Core library source files
set(CORE_SOURCES
  src/thread_pool.cpp
  src/launch_description.cpp
  src/action.cpp
  src/launch_service.cpp
  src/actions/execute_process.cpp
  src/actions/declare_launch_argument.cpp
  src/substitutions/launch_configuration.cpp
  src/substitutions/environment_variable.cpp
  src/conditions/if_condition.cpp
  src/conditions/launch_configuration_equals.cpp
)

# Create library
add_library(cpp_launch_lib ${CORE_SOURCES})
target_link_libraries(cpp_launch_lib ${CMAKE_THREAD_LIBS_INIT})

# Executable
add_executable(cpp_launch_test 
  examples/example_basic.cpp
)
target_link_libraries(cpp_launch_test cpp_launch_lib)

# Simple test executable
add_executable(simple_test 
  test/simple_test.cpp
)
target_link_libraries(simple_test cpp_launch_lib)

enable_testing()
add_test(NAME SimpleTest COMMAND simple_test)

# Print status
message(STATUS "cpp_launch_test configured successfully")
message(STATUS "Build with: cmake --build .")
